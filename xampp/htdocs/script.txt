function sendMessage() {
  var userInput = document.getElementById("userInput");
  var chatbox = document.getElementById("chatbox");
  
  // Benutzereingabe hinzufügen
  var userMessage = document.createElement("p");
  userMessage.innerHTML = "<strong>Du:</strong> " + userInput.value;
  chatbox.appendChild(userMessage);
  
  // Hier kannst du den Benutzereingabe an deinen Chatbot senden und die Antwort empfangen
  
  // Beispielantwort des Chatbots
  var botMessage = document.createElement("p");
  botMessage.innerHTML = "<strong>Chatbot:</strong> Das ist die Antwort des Chatbots.";
  chatbox.appendChild(botMessage);
  
  // Benutzereingabe löschen
  userInput.value = "";
  
  // Automatisch zum Ende der Chatbox scrollen
  chatbox.scrollTop = chatbox.scrollHeight;
}
